/**
 * @fileoverview GoogleLoginButton component for OAuth authentication.
 * Provides Google Sign-In functionality for user authentication.
 */

import React from 'react';
import { GoogleLogin } from '@react-oauth/google';
import '../styles/googleLoginButton.css';
import { getImageForItem } from '../assets/utils/imageMapper';

/**
 * GoogleLoginButton component - Initiates Google OAuth flow.
 * 
 * @param {Object} props - Component props
 * @param {string} [props.returnTo='/'] - URL to return to after login
 * @param {number} [props.functionality=0] - Role/functionality ID for login
 * @returns {JSX.Element} Google sign-in button
 */
export default function GoogleLoginButton({ returnTo = '/', functionality = 0 }) {
  const apiUrl = (import.meta.env.VITE_API_URL || 'http://localhost:8080').replace(/\/$/, ''); // Remove trailing slash
  return (
    
    <button className="google-button" onClick={() => {
        window.location.href = `${apiUrl}/auth/google?returnTo=${encodeURIComponent(returnTo)}&functionality=${functionality}`;
        }}>
        <img className='img' src={getImageForItem("google")} alt="Google" />
        Sign in with Google
    </button>
    
  ); 
}
